-------------------------------------------------
üìå WIDGETS PRINCIPALES (Clases, atributos, m√©todos)
-------------------------------------------------

üîπ Label (Etiqueta)
tk.Label(parent, opciones)
Atributos:
    text="Texto"
    font=("Arial", 14, "bold")
    fg="red", bg="black"
    padx=10, pady=5
    width=20, height=2
    anchor="w"   (alineaci√≥n: n, s, e, w, center)
M√©todos:
    .config(text="Nuevo texto")
    .cget("text")   -> obtener valor de un atributo
    .pack(), .grid(), .place()

üîπ Button (Bot√≥n)
tk.Button(parent, opciones)
Atributos:
    text="Click"
    command=funcion
    font=("Helvetica", 12)
    fg="white", bg="blue"
    width=15, height=2
    state="disabled"  (o "normal")
M√©todos:
    .config(state="normal")
    .invoke()   -> simula un click
    .flash()    -> parpadeo
Eventos:
    bind("<Button-1>", funcion)   # click izquierdo

üîπ Entry (Caja de texto una l√≠nea)
tk.Entry(parent, opciones)
Atributos:
    textvariable=var
    show="*"   (ocultar texto como password)
    justify="left/center/right"
    width=30
M√©todos:
    .get()
    .insert(pos, "texto")
    .delete(0, tk.END)
    .config(state="readonly")

üîπ Text (Caja multilinea)
tk.Text(parent, opciones)
Atributos:
    height=10, width=40
    wrap="word" (ajuste por palabra)
M√©todos:
    .get("1.0", "end-1c")    -> obtener texto
    .insert("end", "texto")
    .delete("1.0", "end")

üîπ Checkbutton (Casilla de verificaci√≥n)
tk.Checkbutton(parent, opciones)
Atributos:
    text="Aceptar"
    variable=var (tk.IntVar() o tk.BooleanVar())
    onvalue=1, offvalue=0
M√©todos:
    .select(), .deselect(), .toggle()

üîπ Radiobutton (Botones de opci√≥n)
tk.Radiobutton(parent, opciones)
Atributos:
    text="Opci√≥n"
    variable=var
    value=1
    indicatoron=True (o False para estilo bot√≥n)
M√©todos:
    .select()

üîπ Listbox (Lista de √≠tems)
tk.Listbox(parent, opciones)
Atributos:
    height=5, selectmode="single" (o "multiple", "extended")
M√©todos:
    .insert(index, "texto")
    .delete(index)
    .get(index)
    .curselection()

üîπ Frame (Contenedor)
tk.Frame(parent, opciones)
Atributos:
    bg="gray", width=200, height=200
M√©todos:
    .pack(), .grid()
Se usa para organizar widgets dentro.

üîπ Canvas (Dibujos)
tk.Canvas(parent, opciones)
Atributos:
    width=200, height=200, bg="white"
M√©todos:
    .create_line(x1,y1,x2,y2, fill="red", width=2)
    .create_rectangle(x1,y1,x2,y2, fill="blue")
    .create_oval(x1,y1,x2,y2, outline="green")
    .create_text(x,y, text="Hola")

üîπ Menu
menu_bar = tk.Menu(root)

sub_menu = tk.Menu(menu_bar, tearoff=0)

menu_bar.add_cascade(label="Archivo", menu=sub_menu)

    fill="x/y/both"

    x=50, y=100
    relx=0.5, rely=0.5 (posici√≥n relativa)
